#include <iostream>
using namespace std;
int main() {

int a[4]={1,2,3,4};
int b[5]={1,2,3,6,9};
int e[10];
int c=4; 
int d=5;
int i ,j ;
int f=c;
int temp=0;
  
  int counter = 0 ; 

  
  for(int i = 0; i!=5;i++){
   e[i]=a[i]; 
  }

for(int i=0 ;i!=d;i++ ){
counter=0 ;

for(int j=0; j!=c && counter!=1;j++){
if(b[i]==a[j]){
  counter=1;
}
if(counter!=1){
  e[f++]=b[i];
}

}
}

for(int i =0 ; i!=f;i++){
for(int j=i+1 ; j!=f;j++){
  if(e[i]==e[j]){
    e[i]=0;
  }

}
}


for(i=0;i<f;i++)
	{		
		for(j=i+1;j<f;j++)
		{
			if(e[i]==0)
			{
				temp =e[i];
				e[i]=e[j];
				e[j]=temp;
			}
		}
	}

	cout<<" A u B = {";
for(int i =0 ; i!=f;i++){

  while(e[i]!=0){
    cout<<e[i]<<", ";

  }
